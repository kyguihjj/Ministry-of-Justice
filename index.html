/* ---------- اصوات بسيطة (WebAudio API) ---------- */

let audioEnabled = true;

const soundToggle = document.getElementById('soundToggle');

soundToggle.addEventListener('change', e=> audioEnabled = e.target.checked);

 

function playTone(freq, duration=0.08, type='sine', gain=0.08){

  if(!audioEnabled) return;

  try{

    const ctx = window.__audioCtx || (window.__audioCtx = new (window.AudioContext || window.webkitAudioContext)());

    const o = ctx.createOscillator();

    const g = ctx.createGain();

    o.type = type;

    o.frequency.value = freq;

    g.gain.value = gain;

    o.connect(g); g.connect(ctx.destination);

    o.start();

    g.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + duration);

    setTimeout(()=>{ try{ o.stop(); }catch(e){} }, duration*1000 + 20);

  }catch(e){}

}

function playClick(){ playTone(650,0.05,'triangle',0.04); }

function playBig(){ playTone(340,0.16,'sine',0.18); playTone(880,0.12,'sine',0.06); }

 

/* ---------- ارسال Embed الى Discord Webhook ---------- */

async function sendEmbedToWebhook(webhookUrl, embed){

  const payload = { embeds: [embed] };

  try{

    // محاولة الارسال مباشرة

    const res = await fetch(webhookUrl, {

      method: 'POST',

      headers: { 'Content-Type': 'application/json' },

      body: JSON.stringify(payload)

    });

    if(!res.ok) throw new Error('Response not OK: ' + res.status);

    return { ok:true };

  }catch(err){

    console.warn('Direct webhook failed (CORS or network):', err);

    return { ok:false, error: err.message, payload };

  }

}
